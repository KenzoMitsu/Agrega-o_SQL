
--Atividade 1--
SELECT SUM(M.ESTOQUE)
	FROM MEDICAMENTOS m

--Atividade 2--
SELECT AVG(EXTRACT(YEAR FROM CURRENT_DATE) - EXTRACT(YEAR FROM t.DATA_NASCIMENTO)) AS MEDIA_IDADE
FROM TUTOR t

--Atividade 3--
SELECT MIN(m.PRECO)
	FROM MEDICAMENTOS m

--Atividade 4--
SELECT MAX(p.PESO)
	FROM PACIENTE p

--Atividade 5--
SELECT COUNT(*) AS TOTAL_TUTORES
	FROM TUTOR

--Atividade 6--
SELECT ESTOQUE * PRECO AS VALOR_TOTAL
	FROM MEDICAMENTOS m

--Atividade 7--
SELECT ID_RECEITA, SUM(TOTAL_ITEM) AS VALOR_TOTAL
	FROM ITENSRECEITA
   GROUP BY ID_RECEITA
   HAVING SUM(TOTAL_ITEM) > 100
